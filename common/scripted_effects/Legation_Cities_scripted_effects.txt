increase_currency_reserve = {
	custom_effect_tooltip = LEC_currency_up
	hidden_effect = {
		# do not go above 10
		if = {
			limit = {
				NOT = {
					has_country_flag = {
						flag = LEC_currency_reserves
						value > 9
					}
				}
			}
			modify_country_flag = {
				flag = LEC_currency_reserves
				value = 1
			}
			if = {
				limit = {
					has_country_flag = {
						flag = LEC_currency_reserves
						value > 1
					}
					has_country_flag = {
						flag = LEC_currency_reserves
						value < 3
					}
				}
				swap_ideas = {
					remove_idea = LEC_currency_1
					add_idea = LEC_currency_2
				}
			}
			if = {
				limit = {
					has_country_flag = {
						flag = LEC_currency_reserves
						value > 2
					}
					has_country_flag = {
						flag = LEC_currency_reserves
						value < 4
					}
				}
				swap_ideas = {
					remove_idea = LEC_currency_2
					add_idea = LEC_currency_3
				}
			}
			if = {
				limit = {
					has_country_flag = {
						flag = LEC_currency_reserves
						value > 3
					}
					has_country_flag = {
						flag = LEC_currency_reserves
						value < 5
					}
				}
				swap_ideas = {
					remove_idea = LEC_currency_3
					add_idea = LEC_currency_4
				}
			}
			if = {
				limit = {
					has_country_flag = {
						flag = LEC_currency_reserves
						value > 4
					}
					has_country_flag = {
						flag = LEC_currency_reserves
						value < 6
					}
				}
				swap_ideas = {
					remove_idea = LEC_currency_4
					add_idea = LEC_currency_5
				}
			}
			if = {
				limit = {
					has_country_flag = {
						flag = LEC_currency_reserves
						value > 5
					}
					has_country_flag = {
						flag = LEC_currency_reserves
						value < 7
					}
				}
				swap_ideas = {
					remove_idea = LEC_currency_5
					add_idea = LEC_currency_6
				}
			}
			if = {
				limit = {
					has_country_flag = {
						flag = LEC_currency_reserves
						value > 6
					}
					has_country_flag = {
						flag = LEC_currency_reserves
						value < 8
					}
				}
				swap_ideas = {
					remove_idea = LEC_currency_6
					add_idea = LEC_currency_7
				}
			}
			if = {
				limit = {
					has_country_flag = {
						flag = LEC_currency_reserves
						value > 7
					}
					has_country_flag = {
						flag = LEC_currency_reserves
						value < 9
					}
				}
				swap_ideas = {
					remove_idea = LEC_currency_7
					add_idea = LEC_currency_8
				}
			}
			if = {
				limit = {
					has_country_flag = {
						flag = LEC_currency_reserves
						value > 8
					}
					has_country_flag = {
						flag = LEC_currency_reserves
						value < 10
					}
				}
				swap_ideas = {
					remove_idea = LEC_currency_8
					add_idea = LEC_currency_9
				}
			}
			if = {
				limit = {
					has_country_flag = {
						flag = LEC_currency_reserves
						value > 9
					}
					has_country_flag = {
						flag = LEC_currency_reserves
						value < 11
					}
				}
				swap_ideas = {
					remove_idea = LEC_currency_9
					add_idea = LEC_currency_10
				}
			}
		}
	}
}

decrease_currency_reserve = {
	custom_effect_tooltip = LEC_currency_down
	hidden_effect = {
		# do not go under 1
		if = {
			limit = {
				NOT = {
					has_country_flag = {
						flag = LEC_currency_reserves
						value < 1
					}
				}
			}

			modify_country_flag = {
				flag = LEC_currency_reserves
				value = -1
			}

			if = {
				limit = {
					has_country_flag = {
						flag = LEC_currency_reserves
						value > 0
					}
					has_country_flag = {
						flag = LEC_currency_reserves
						value < 2
					}
				}
				swap_ideas = {
					remove_idea = LEC_currency_2
					add_idea = LEC_currency_1
				}
			}
			if = {
				limit = {
					has_country_flag = {
						flag = LEC_currency_reserves
						value > 1
					}
					has_country_flag = {
						flag = LEC_currency_reserves
						value < 3
					}
				}
				swap_ideas = {
					remove_idea = LEC_currency_3
					add_idea = LEC_currency_2
				}
			}
			if = {
				limit = {
					has_country_flag = {
						flag = LEC_currency_reserves
						value > 2
					}
					has_country_flag = {
						flag = LEC_currency_reserves
						value < 4
					}
				}
				swap_ideas = {
					remove_idea = LEC_currency_4
					add_idea = LEC_currency_3
				}
			}
			if = {
				limit = {
					has_country_flag = {
						flag = LEC_currency_reserves
						value > 3
					}
					has_country_flag = {
						flag = LEC_currency_reserves
						value < 5
					}
				}
				swap_ideas = {
					remove_idea = LEC_currency_5
					add_idea = LEC_currency_4
				}
			}
			if = {
				limit = {
					has_country_flag = {
						flag = LEC_currency_reserves
						value > 4
					}
					has_country_flag = {
						flag = LEC_currency_reserves
						value < 6
					}
				}
				swap_ideas = {
					remove_idea = LEC_currency_6
					add_idea = LEC_currency_5
				}
			}
			if = {
				limit = {
					has_country_flag = {
						flag = LEC_currency_reserves
						value > 5
					}
					has_country_flag = {
						flag = LEC_currency_reserves
						value < 7
					}
				}
				swap_ideas = {
					remove_idea = LEC_currency_7
					add_idea = LEC_currency_6
				}
			}
			if = {
				limit = {
					has_country_flag = {
						flag = LEC_currency_reserves
						value > 6
					}
					has_country_flag = {
						flag = LEC_currency_reserves
						value < 8
					}
				}
				swap_ideas = {
					remove_idea = LEC_currency_8
					add_idea = LEC_currency_7
				}
			}
			if = {
				limit = {
					has_country_flag = {
						flag = LEC_currency_reserves
						value > 7
					}
					has_country_flag = {
						flag = LEC_currency_reserves
						value < 9
					}
				}
				swap_ideas = {
					remove_idea = LEC_currency_9
					add_idea = LEC_currency_8
				}
			}
			if = {
				limit = {
					has_country_flag = {
						flag = LEC_currency_reserves
						value > 8
					}
					has_country_flag = {
						flag = LEC_currency_reserves
						value < 10
					}
				}
				swap_ideas = {
					remove_idea = LEC_currency_10
					add_idea = LEC_currency_9
				}
			}
		}
	}
}

lec_british_rise = {
	if = {
		limit = {
			NOT = {
				has_idea = LEC_british_up
			}
		}
		add_timed_idea = {
			idea = LEC_british_up
			days = 50
		}
	}
	else = {
		modify_timed_idea = {
			idea = LEC_british_up
			days = 50
		}
	}
}

LEC_exile_georg_wetzell = {
	if = {
		limit = {
			country_exists = LEP
			LEP = { has_country_flag = LEP_AOG_rule }
		}
		random_unit_leader = {
			limit = { has_id = 1354 }
			set_nationality = LEP
		}
	}
	else_if = {
		limit = { country_exists = GEA }
		random_unit_leader = {
			limit = { has_id = 1354 }
			set_nationality = GEA
		}
	}
	else_if = {
		limit = { country_exists = GER }
		random_unit_leader = {
			limit = { has_id = 1354 }
			set_nationality = GER
		}
	}
	else_if = {
		limit = {
			has_global_flag = GER_became_GEX
			country_exists = GEX
		}
		random_unit_leader = {
			limit = { has_id = 1354 }
			set_nationality = GEX
		}
	}
	else_if = {
		limit = { country_exists = MAF }
		random_unit_leader = {
			limit = { has_id = 1354 }
			set_nationality = MAF
		}
	}
	else_if = {
		limit = { country_exists = COG }
		random_unit_leader = {
			limit = { has_id = 1354 }
			set_nationality = COG
		}
	}
	else_if = {

			limit = { has_id = 1354 }
